    <script>
        // 1. Scroll Reveal Animation
        document.addEventListener('DOMContentLoaded', () => {
            const reveals = document.querySelectorAll('.reveal');
            const mainEl = document.querySelector('main');
            
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        entry.target.classList.add('active');
                    }
                });
            }, { root: mainEl, threshold: 0.1 });
            
            reveals.forEach(el => observer.observe(el));
        });

        // 2. PC Background Canvas Animation (Confetti Effect)
        const canvas = document.getElementById('bg-canvas');
        const ctx = canvas.getContext('2d');
        
        let width, height;
        let confetti = [];
        const colors = ['#06b6d4', '#f43f5e', '#8b5cf6', '#f59e0b', '#10b981']; // Jeju Party Colors

        function resize() {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        }
        
        class ConfettiPiece {
            constructor() {
                this.reset();
                this.y = Math.random() * height; // Start scattered
            }
            
            reset() {
                this.x = Math.random() * width;
                this.y = -20;
                this.size = (Math.random() * 0.5 + 0.5) * 10;
                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.speedY = Math.random() * 2 + 1;
                this.speedX = Math.random() * 2 - 1;
                this.rotation = Math.random() * 360;
                this.rotationSpeed = Math.random() * 2 - 1;
            }
            
            update() {
                this.y += this.speedY;
                this.x += this.speedX;
                this.rotation += this.rotationSpeed;

                if (this.y > height) {
                    this.reset();
                }
            }
            
            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.rotation * Math.PI / 180);
                ctx.fillStyle = this.color;
                ctx.globalAlpha = 0.8;
                ctx.fillRect(-this.size / 2, -this.size / 2, this.size, this.size);
                ctx.restore();
            }
        }
        
        function initConfetti() {
            confetti = [];
            for (let i = 0; i < 120; i++) {
                confetti.push(new ConfettiPiece());
            }
        }
        
        function animate() {
            ctx.clearRect(0, 0, width, height);
            
            // Draw gradient background
            const gradient = ctx.createRadialGradient(width/2, height/2, 0, width/2, height/2, width);
            gradient.addColorStop(0, '#111827');
            gradient.addColorStop(1, '#000000');
            ctx.fillStyle = gradient;
            ctx.fillRect(0, 0, width, height);
            
            confetti.forEach(p => {
                p.update();
                p.draw();
            });
            
            requestAnimationFrame(animate);
        }
        
        window.addEventListener('resize', resize);
        resize();
        initConfetti();
        animate();
    </script>
